create table public.art_docs (
  id uuid not null default extensions.uuid_generate_v4 (),
  firestore_id text null,
  name text null,
  title text null,
  description text null,
  categorie text null,
  size text null,
  height integer null,
  width integer null,
  is_3d boolean null,
  is_premuim boolean null,
  exported_times integer null,
  status text null,
  preview_realized_art text null,
  generated_preview_url text null,
  video text null,
  thumbnails jsonb null,
  tags text[] null,
  reviews jsonb null,
  created_at timestamp without time zone null,
  updated_at timestamp without time zone null,
  constraint art_docs_pkey primary key (id)
) TABLESPACE pg_default;


create table public.art_docs_faces (
  id text not null,
  art_doc_id uuid null,
  name text null,
  "backgroundColor" text null,
  "canvasContent" jsonb null,
  height integer null,
  width integer null,
  preview text null,
  created_at timestamp without time zone null default now(),
  size text null,
  firestore_id text null,
  updated_at timestamp without time zone null,
  side text null default 'recto'::text,
  constraint art_docs_pages_pkey primary key (id),
  constraint art_docs_pages_art_doc_id_fkey foreign KEY (art_doc_id) references art_docs (id) on delete CASCADE,
  constraint fk_art_doc foreign KEY (art_doc_id) references art_docs (id) on delete CASCADE
) TABLESPACE pg_default;

create table public.exports (
  id uuid not null default extensions.uuid_generate_v4 (),
  user_id uuid not null,
  art_doc_id uuid not null,
  format text not null,
  created_at timestamp without time zone null default now(),
  constraint exports_pkey primary key (id),
  constraint exports_art_doc_id_fkey foreign KEY (art_doc_id) references art_docs (id) on delete CASCADE,
  constraint exports_user_id_fkey foreign KEY (user_id) references profiles (id) on delete CASCADE
) TABLESPACE pg_default;

create table public.profiles (
  id uuid not null,
  full_name text null,
  email text null,
  avatar_url text null,
  app_role text null default 'guest-user'::text,
  created_at timestamp without time zone null default now(),
  has_password boolean null default false,
  updated_at timestamp with time zone null,
  constraint profiles_pkey primary key (id),
  constraint profiles_id_fkey foreign KEY (id) references auth.users (id) on delete CASCADE,
  constraint profiles_app_role_check check (
    (
      app_role = any (
        array[
          'user'::text,
          'admin'::text,
          'anonymous-user'::text,
          'guest-user'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;
